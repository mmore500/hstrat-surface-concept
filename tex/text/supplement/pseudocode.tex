\section{Pseudocode Helper Functions}

\begin{algorithmic}
  \Function{BitCount}{$x$}
    \LComment{Equivalent \texttt{std::popcount}}
    \State \Return $|\{ n \in \mathbb{N} : x \bmod 2^n = 0 \}|$
  \EndFunction
  \Statex
  \Function{BitFloor}{$x$}
    \State \Return \Call{LeftShift}{1, \;\; \Call{BitLength}{$x$} - 1}
  \EndFunction
  \Statex
  \Function{BitFloorSafe}{$x$}
    \LComment{Equivalent \texttt{std::bit\_floor}}
    \State \Return \Call{BitFloor}{$x$} \text{ if } $x > 0$ \text{ else } 0
  \EndFunction
  \Statex
  \Function{BitLength}{$x$}
    \LComment{Equivalent \texttt{std::bit\_width}}
    \State \Return $\left\lfloor\log_{2}(x)\right\rfloor + 1  \text{ if } x \text{ else } 0$
  \EndFunction
  \Statex
  \Function{Bool2Int}{$x$}
    \LComment{Equivalent \texttt{static\_cast<bool>}}
    \State \Return 1 \text{ if } $x$ \text{ else } 0
  \EndFunction
  \Statex
  \Function{CountTrailingZeros}{$x$}
    \LComment{Equivalent \texttt{std::countr\_zero}}
    \State \Return $\max \{ n \in \mathbb{N} : x \bmod 2^n = 0 \}$
  \EndFunction
  \Statex
  \Function{ElvisOp}{$x, \;\; y$}
    \LComment{Equivalent \texttt{?:}}
    \State \Return $x \text{ if } x \neq 0 \text{ else } y$
  \EndFunction
  \Statex
  \Function{LeftShift}{$x, \;\; n$}
    \State \Return $2^n x$
  \EndFunction
  \Statex
  \Function{ModPow2}{$x, \;\; n$}
    \LComment{Equivalent $x \bmod n$, requires $n \in \{2^{\mathbb{N}}\}$}
    \State \Return $x \& (n - 1)$
  \EndFunction
  \Statex
  \Function{RightShift}{$x,  \;\;n$}
    \State \Return $\left\lfloor x/2^n \right\rfloor$
  \EndFunction
  \Statex
\end{algorithmic}
