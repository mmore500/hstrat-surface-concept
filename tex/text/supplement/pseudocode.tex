\section{Pseudocode Helper Functions}

\begin{algorithmic}
  \Function{BitFloor}{$x$}
    \State \Return \Call{LeftShift}{1, \Call{BitLength}{$x$} - 1}
  \EndFunction
  \Statex
  \Function{BitFloorSafe}{$x$}
    \State \Return \Call{BitFloor}{$x$} \text{ if } $x > 0$ \text{ else } 0
  \EndFunction
  \Statex
  \Function{BitLength}{$x$}
    \State \Return $\left\lfloor\log_{2}(x)\right\rfloor + 1  \text{ if } x \text{ else } 0$
  \EndFunction
  \Statex
  \Function{Bool2Int}{$x$}
    \State \Return 1 \text{ if } $x$ \text{ else } 0
  \EndFunction
  \Statex
  \Function{CountTrailingZeros}{$x$}
    \State \Return $\max \{ n \in \mathbb{N} : x \bmod 2^n = 0 \}$
  \EndFunction
  \Statex
  \Function{LeftShift}{$x, n$}
    \State \Return $2^n x$
  \EndFunction
  \Statex
  \Function{ModPow2}{$x, n$}
    \LComment{Equivalent $x \bmod n$, assuming $n \in 2^{\mathbb{N}}$}
    \State \Return $x \& (n - 1)$
  \EndFunction
  \Statex
  \Function{RightShift}{$x, n$}
    \State \Return $\left\lfloor x/2^n \right\rfloor$
  \EndFunction
  \Statex
\end{algorithmic}
